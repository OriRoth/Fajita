Program_Definition  ->  program Identifier Parameters ; Definitions Body.
Identifier          ->  id.
Body                ->  begin end.

Definitions  -> Labels Constants Types Variables Nested.

Labels      -> | label Label_Declaration Optional_Labels . 
Constants   -> | const Constant Optional_Constants .
Types       -> | type  Type_Definition Optional_Types .
Variables   -> | var   Variable Optional_Variables .  

Label_Declaration    -> Semi_Thing.
Constant             -> Semi_Thing.
Type_Definition      -> Semi_Thing.
Variable             -> Semi_Thing.
Semi_Thing           -> Thing ;   .
Thing                -> .

Optional_Labels     ->  |  Label_Declaration Optional_Labels.
Optional_Constants  ->  |  Constant Optional_Constants.
Optional_Types      ->  |  Type_Definition Optional_Types.
Optional_Variables  ->  |  Variable Optional_Variables.

Nested -> | Module Nested.
Module  -> Function_Definition | Procedure_Definition.

Procedure_Definition -> procedure Identifier Parameters ; Definitions ( Body.
Function_Definition  -> function Identifier Parameters : Identifier ; Definitions Body.
Parameters           -> | ().
